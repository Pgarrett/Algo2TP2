\section{Módulo Lista Enlazada Acotada(puntero($\alpha$))}

El módulo Lista Enlazada Acotada provee una lista por posiciones en el que se puede definir, borrar, y testear si hay un valor en una posicion en tiempo $O(1)$. 

%El principal costo de paga al crear la estructura, dado de cuesta tiempo lineal $ancho$ por $largo$.

\begin{Interfaz}
  
  \textbf{parámetros formales}\hangindent=2\parindent\\
  \parbox{1.7cm}{\textbf{géneros}} $\alpha$\\
  \parbox[t]{1.7cm}{\textbf{función}}\parbox[t]{\textwidth-2\parindent-1.7cm}{%
    \InterfazFuncion{Copiar}{\In{a}{$\alpha$}}{$\alpha$}
    {$res \igobs a$}
    [$\Theta(copy(a))$]
    [función de copia de $\alpha$'s]
    %\TipoFuncion{Copiar}{\In{a}{$\alpha$}}{$\alpha$} \qquad función de copia, con costo temporal $\Theta(copy(a))$.
  }

  \textbf{se explica con}: \tadNombre{Secuencia$(\alpha)$}, \tadNombre{Iterador Bidireccional($\alpha$)}.

  \textbf{géneros}: \TipoVariable{listaAc$(\alpha)$}, \TipoVariable{itListaAc($\alpha$)}.

  \Titulo{Operaciones básicas de lista}

  \InterfazFuncion{Vacía}{\In{lon}{Nat}, \In{an}{Nat}}{listaAc$(\alpha)$}%
  [$an > 0 \land lon > 0$]
  {$long(res) = lon$}%
  [$\Theta(lon)$]
  [Genera una lista de largo lon, donde cada nodo es un puntero a null.]

  \InterfazFuncion{Insertar}{\Inout{l}{listaAc($\alpha$)}, \In{d}{$\alpha$}, \In{x}{Nat}, \In{y}{Nat}}{itListaAc($\alpha$)}
  [$f(x,y) < long(l) \land l \igobs l_0 \land \neg$est?($l_0$, d)]
  {est?($l_0$, d)}
  [$\Theta(1)$]
  [agrega el elemento $d$ en la posicion i-esima de la lista.  Retorna un iterador a $l$, de forma tal que Siguiente devuelva $d$.]
 

  \InterfazFuncion{Longitud}{\In{l}{listaAc($\alpha$)}}{Nat}
  {$res \igobs long(l)$}
  [$\Theta(1)$]
  [devuelve la longitud de la lista $l$]

  \InterfazFuncion{HayAlguienAhi?}{\In{l}{listaAc($\alpha$)}, \In{x}{Nat}, \In{y}{Nat}}{bool}
  [$f(x,y) < long(l)$]
  {res $\igobs$ est?(l, f(x,y))}
  [$\Theta(1)$]
  [devuelve \texttt{true} si y sólo si hay algo un elemento que no apunta a null en esa posicion]

  \InterfazFuncion{Iesimo}{\Inout{l}{listaAc($\alpha$)}, \In{x}{Nat}, \In{y}{Nat}}{$\alpha$}
  [$f(x,y) < long(l)$]
  {alias($res \igobs iesimo(l,f(x,y))$)}
  [$\Theta(1)$]
  [devuelve el elemento en la posicion f(x,y)]

  \InterfazFuncion{BorrarIesimo}{\Inout{l}{listaAc($\alpha$)}, \In{x}{Nat}, \In{y}{Nat}}{}
  [$f(x,y) < long(l) \land l \igobs l_0$]
  {alias($secuSuby(l_0, l)$)}
  [$\Theta(1)$]
  [elimina el elemento en la posicion f(x,y)]


\end{Interfaz}
